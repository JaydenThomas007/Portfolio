# DHCP ranges (forwarded via helper)

dhcp-range=set:vlan20,172.16.0.68,172.16.0.78,255.255.255.240,12h
dhcp-range=set:vlan30,172.16.0.101,172.16.0.102,255.255.255.248,12h
# Gateway for each VLAN 
dhcp-option=tag:VLAN10,option:router,172.16.0.1

# Authoritative server

# VLAN-specific DNS 
dhcp-option=option:dns-server,172.16.0.100
